1. Create SQL migration for RLS policies. Since we're using Prisma with Supabase, create a raw SQL migration file:

   Create file: prisma/migrations/20260204_rls_policies/migration.sql
   ```sql
   -- Enable RLS on all tables
   ALTER TABLE "Worker" ENABLE ROW LEVEL SECURITY;
   ALTER TABLE "Service" ENABLE ROW LEVEL SECURITY;
   ALTER TABLE "Resource" ENABLE ROW LEVEL SECURITY;
   ALTER TABLE "Customer" ENABLE ROW LEVEL SECURITY;
   ALTER TABLE "Booking" ENABLE ROW LEVEL SECURITY;
   ALTER TABLE "WorkerSchedule" ENABLE ROW LEVEL SECURITY;

   -- Workers: Public read, admin write
   CREATE POLICY "Workers are viewable by everyone"
     ON "Worker" FOR SELECT
     USING (true);

   CREATE POLICY "Workers are manageable by admin"
     ON "Worker" FOR ALL
     USING (current_setting('app.role', true) = 'admin');

   -- Services: Public read, admin write
   CREATE POLICY "Services are viewable by everyone"
     ON "Service" FOR SELECT
     USING (true);

   CREATE POLICY "Services are manageable by admin"
     ON "Service" FOR ALL
     USING (current_setting('app.role', true) = 'admin');

   -- Resources: Public read, admin write
   CREATE POLICY "Resources are viewable by everyone"
     ON "Resource" FOR SELECT
     USING (true);

   CREATE POLICY "Resources are manageable by admin"
     ON "Resource" FOR ALL
     USING (current_setting('app.role', true) = 'admin');

   -- Customers: Own record only (by email or id in JWT), admin all
   CREATE POLICY "Customers can view own record"
     ON "Customer" FOR SELECT
     USING (
       id = current_setting('app.customer_id', true)::uuid
       OR current_setting('app.role', true) = 'admin'
     );

   CREATE POLICY "Customers can update own record"
     ON "Customer" FOR UPDATE
     USING (id = current_setting('app.customer_id', true)::uuid);

   CREATE POLICY "Admin can manage all customers"
     ON "Customer" FOR ALL
     USING (current_setting('app.role', true) = 'admin');

   -- Bookings: Own bookings only, admin all
   CREATE POLICY "Customers can view own bookings"
     ON "Booking" FOR SELECT
     USING (
       "customerId" = current_setting('app.customer_id', true)::uuid
       OR current_setting('app.role', true) = 'admin'
     );

   CREATE POLICY "Customers can create own bookings"
     ON "Booking" FOR INSERT
     WITH CHECK ("customerId" = current_setting('app.customer_id', true)::uuid);

   CREATE POLICY "Admin can manage all bookings"
     ON "Booking" FOR ALL
     USING (current_setting('app.role', true) = 'admin');

   -- WorkerSchedule: Public read, admin write
   CREATE POLICY "WorkerSchedule viewable by everyone"
     ON "WorkerSchedule" FOR SELECT
     USING (true);

   CREATE POLICY "WorkerSchedule manageable by admin"
     ON "WorkerSchedule" FOR ALL
     USING (current_setting('app.role', true) = 'admin');
   ```

2. Note: These policies use PostgreSQL session variables (current_setting) that need to be set before queries. For MVP, this provides defense-in-depth but Prisma operations happen with full access. Full RLS enforcement requires setting session variables per request.

3. Create a utility to set RLS context (for future use):
   Create src/lib/db/rls-context.ts:
   ```typescript
   import { prisma } from './client'

   export async function withRLSContext<T>(
     context: { customerId?: string; role?: 'admin' | 'user' },
     operation: () => Promise<T>
   ): Promise<T> {
     // Set session variables for RLS
     await prisma.$executeRaw`SELECT set_config('app.customer_id', ${context.customerId || ''}, true)`
     await prisma.$executeRaw`SELECT set_config('app.role', ${context.role || 'user'}, true)`

     return operation()
   }
   ```

2. Create src/lib/rate-limit.ts:
   ```typescript
   import { Ratelimit } from '@upstash/ratelimit'
   import { Redis } from '@upstash/redis'

   // Use in-memory store for development if no Upstash configured
   const redis = process.env.UPSTASH_REDIS_REST_URL
     ? Redis.fromEnv()
     : null

   // Booking endpoint: 10 requests per minute per IP
   export const bookingRateLimit = redis
     ? new Ratelimit({
         redis,
         limiter: Ratelimit.slidingWindow(10, '1 m'),
         analytics: true,
         prefix: 'synq:booking',
       })
     : null

   // General API: 100 requests per minute per IP
   export const apiRateLimit = redis
     ? new Ratelimit({
         redis,
         limiter: Ratelimit.slidingWindow(100, '1 m'),
         analytics: true,
         prefix: 'synq:api',
       })
     : null

   export interface RateLimitResult {
     success: boolean
     limit: number
     remaining: number
     reset: number
   }

   export async function checkBookingRateLimit(ip: string): Promise<RateLimitResult> {
     if (!bookingRateLimit) {
       // Development mode: no rate limiting
       return { success: true, limit: 10, remaining: 10, reset: Date.now() + 60000 }
     }

     const result = await bookingRateLimit.limit(ip)
     return {
       success: result.success,
       limit: result.limit,
       remaining: result.remaining,
       reset: result.reset,
     }
   }
   ```

3. Update middleware.ts to add rate limiting:
   ```typescript
   // Add to existing middleware:
   import { checkBookingRateLimit } from '@/lib/rate-limit'

   // In middleware function:
   if (pathname.includes('/api/booking') && request.method === 'POST') {
     const ip = request.ip ?? request.headers.get('x-forwarded-for') ?? '127.0.0.1'
     const result = await checkBookingRateLimit(ip)

     if (!result.success) {
       return NextResponse.json(
         { error: 'Too many booking requests. Please try again later.' },
         {
           status: 429,
           headers: {
             'X-RateLimit-Limit': result.limit.toString(),
             'X-RateLimit-Remaining': result.remaining.toString(),
             'X-RateLimit-Reset': result.reset.toString(),
             'Retry-After': '60',
           },
         }
       )
     }
   }
   ```
