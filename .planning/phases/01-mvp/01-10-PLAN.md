1. Create src/lib/validations/admin-booking.ts:

   ```typescript
   import { z } from 'zod'

   export const updateBookingSchema = z.object({
     bookingId: z.string().uuid(),
     startsAt: z.coerce.date().optional(),
     workerId: z.string().uuid().optional(),
     status: z.enum(['CONFIRMED', 'CANCELLED', 'NOSHOW']).optional(),
   })

   export const blockTimeSchema = z.object({
     workerId: z.string().uuid(),
     date: z.coerce.date(),
     startTime: z.string().regex(/^\d{2}:\d{2}$/),
     endTime: z.string().regex(/^\d{2}:\d{2}$/),
   })
   ```

2. Create app/actions/admin-booking.ts:
   ```typescript
   'use server'

   import { revalidatePath } from 'next/cache'
   import { prisma } from '@/lib/db/client'
   import { updateBookingSchema, blockTimeSchema } from '@/lib/validations/admin-booking'
   import { getAdminSession } from '@/lib/auth/admin'

   export async function cancelBooking(bookingId: string) {
     const isAdmin = await getAdminSession()
     if (!isAdmin) throw new Error('Unauthorized')

     await prisma.booking.update({
       where: { id: bookingId },
       data: { status: 'CANCELLED' },
     })

     revalidatePath('/admin/dashboard')
     return { success: true }
   }

   export async function updateBooking(formData: FormData) {
     const isAdmin = await getAdminSession()
     if (!isAdmin) throw new Error('Unauthorized')

     const raw = Object.fromEntries(formData)
     const parsed = updateBookingSchema.parse(raw)

     await prisma.booking.update({
       where: { id: parsed.bookingId },
       data: {
         ...(parsed.startsAt && { startsAt: parsed.startsAt }),
         ...(parsed.workerId && { workerId: parsed.workerId }),
         ...(parsed.status && { status: parsed.status }),
       },
     })

     revalidatePath('/admin/dashboard')
     return { success: true }
   }

   export async function blockWorkerTime(formData: FormData) {
     const isAdmin = await getAdminSession()
     if (!isAdmin) throw new Error('Unauthorized')

     const raw = Object.fromEntries(formData)
     const parsed = blockTimeSchema.parse(raw)

     await prisma.workerSchedule.create({
       data: {
         workerId: parsed.workerId,
         specificDate: parsed.date,
         startTime: parsed.startTime,
         endTime: parsed.endTime,
         isAvailable: false, // This marks blocked time
       },
     })

     revalidatePath('/admin/dashboard')
     return { success: true }
   }

   export async function removeBlockedTime(scheduleId: string) {
     const isAdmin = await getAdminSession()
     if (!isAdmin) throw new Error('Unauthorized')

     await prisma.workerSchedule.delete({
       where: { id: scheduleId },
     })

     revalidatePath('/admin/dashboard')
     return { success: true }
   }
   ```

2. Create app/actions/workers.ts:
   ```typescript
   'use server'

   import { revalidatePath } from 'next/cache'
   import { prisma } from '@/lib/db/client'
   import { workerSchema } from '@/lib/validations/worker'
   import { getAdminSession } from '@/lib/auth/admin'

   export async function createWorker(formData: FormData) {
     const isAdmin = await getAdminSession()
     if (!isAdmin) throw new Error('Unauthorized')

     const parsed = workerSchema.parse(Object.fromEntries(formData))
     await prisma.worker.create({ data: parsed })

     revalidatePath('/admin/workers')
     return { success: true }
   }

   export async function updateWorker(formData: FormData) {
     const isAdmin = await getAdminSession()
     if (!isAdmin) throw new Error('Unauthorized')

     const id = formData.get('id') as string
     const parsed = workerSchema.parse(Object.fromEntries(formData))

     await prisma.worker.update({ where: { id }, data: parsed })

     revalidatePath('/admin/workers')
     return { success: true }
   }

   export async function deleteWorker(id: string) {
     const isAdmin = await getAdminSession()
     if (!isAdmin) throw new Error('Unauthorized')

     // Soft delete: set isActive to false
     await prisma.worker.update({
       where: { id },
       data: { isActive: false },
     })

     revalidatePath('/admin/workers')
     return { success: true }
   }
   ```
