   ```typescript
   import { prisma } from '@/lib/db/client'
   import { Prisma } from '@prisma/client'
   import { checkWorkerAvailability, checkResourceAvailability } from './availability.service'
   import { createBookingSchema, BookingInput } from '@/lib/validations/booking'

   const MAX_RETRIES = 3
   const RETRY_DELAY_MS = 100

   export async function createBooking(input: BookingInput) {
     // 1. Validate input with Zod
     const validated = createBookingSchema.parse(input)

     // 2. Get service to calculate end time
     const service = await prisma.service.findUnique({
       where: { id: validated.serviceId },
       select: { duration: true },
     })
     if (!service) throw new Error('Service not found')

     const endsAt = new Date(validated.startsAt.getTime() + service.duration * 60000)

     // 3. Attempt booking with retries for serialization failures
     for (let attempt = 1; attempt <= MAX_RETRIES; attempt++) {
       try {
         return await prisma.$transaction(
           async (tx) => {
             // Check worker availability within transaction
             const workerBookings = await tx.booking.findMany({
               where: {
                 workerId: validated.workerId,
                 status: { in: ['CONFIRMED', 'PENDING'] },
                 OR: [
                   { startsAt: { lte: validated.startsAt }, endsAt: { gt: validated.startsAt } },
                   { startsAt: { lt: endsAt }, endsAt: { gte: endsAt } },
                   { startsAt: { gte: validated.startsAt }, endsAt: { lte: endsAt } },
                 ],
               },
             })

             if (workerBookings.length > 0) {
               throw new Error('Worker not available at this time')
             }

             // Find or verify resource availability
             let resourceId = validated.resourceId
             if (!resourceId) {
               // Auto-assign first available resource
               const availableResource = await findAvailableResource(tx, validated.startsAt, endsAt)
               if (!availableResource) {
                 throw new Error('No resources available at this time')
               }
               resourceId = availableResource.id
             } else {
               // Verify specified resource is available
               const resourceBookings = await tx.booking.findMany({
                 where: {
                   resourceId,
                   status: { in: ['CONFIRMED', 'PENDING'] },
                   OR: [
                     { startsAt: { lte: validated.startsAt }, endsAt: { gt: validated.startsAt } },
                     { startsAt: { lt: endsAt }, endsAt: { gte: endsAt } },
                     { startsAt: { gte: validated.startsAt }, endsAt: { lte: endsAt } },
                   ],
                 },
               })
               if (resourceBookings.length > 0) {
                 throw new Error('Resource not available at this time')
               }
             }

             // Create booking
             return tx.booking.create({
               data: {
                 customerId: validated.customerId,
                 serviceId: validated.serviceId,
                 workerId: validated.workerId,
                 resourceId,
                 startsAt: validated.startsAt,
                 endsAt,
                 status: 'CONFIRMED',
               },
             })
           },
           {
             isolationLevel: Prisma.TransactionIsolationLevel.Serializable,
             timeout: 10000,
           }
         )
       } catch (error) {
         // Retry on serialization failure (P2034)
         if (
           error instanceof Prisma.PrismaClientKnownRequestError &&
           error.code === 'P2034' &&
           attempt < MAX_RETRIES
         ) {
           await new Promise(r => setTimeout(r, RETRY_DELAY_MS * attempt))
           continue
         }
         throw error
       }
     }
   }

   async function findAvailableResource(
     tx: Prisma.TransactionClient,
     startsAt: Date,
     endsAt: Date
   ) {
     const resources = await tx.resource.findMany({
       where: { isActive: true },
     })

     for (const resource of resources) {
       const conflicts = await tx.booking.count({
         where: {
           resourceId: resource.id,
           status: { in: ['CONFIRMED', 'PENDING'] },
           OR: [
             { startsAt: { lte: startsAt }, endsAt: { gt: startsAt } },
             { startsAt: { lt: endsAt }, endsAt: { gte: endsAt } },
             { startsAt: { gte: startsAt }, endsAt: { lte: endsAt } },
           ],
         },
       })
       if (conflicts === 0) return resource
     }
     return null
   }

   export async function cancelBooking(bookingId: string) {
     return prisma.booking.update({
       where: { id: bookingId },
       data: { status: 'CANCELLED' },
     })
   }
   ```
