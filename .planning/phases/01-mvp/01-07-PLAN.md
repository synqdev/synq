2. Create src/lib/services/customer.service.ts:
   ```typescript
   import { prisma } from '@/lib/db/client'
   import { RegisterCustomerInput } from '@/lib/validations/customer'

   export async function findOrCreateCustomer(input: RegisterCustomerInput) {
     // Upsert: Create if not exists, return existing if found by email
     return prisma.customer.upsert({
       where: { email: input.email },
       update: {}, // Don't update existing customer
       create: {
         email: input.email,
         name: input.name,
         phone: input.phone,
         locale: input.locale,
       },
     })
   }
   ```

3. Create app/actions/customer.ts:
   ```typescript
   'use server'

   import { redirect } from 'next/navigation'
   import { cookies } from 'next/headers'
   import { registerCustomerSchema } from '@/lib/validations/customer'
   import { findOrCreateCustomer } from '@/lib/services/customer.service'

   export async function registerCustomer(formData: FormData) {
     const raw = Object.fromEntries(formData)
     const parsed = registerCustomerSchema.safeParse(raw)

     if (!parsed.success) {
       return { error: parsed.error.flatten().fieldErrors }
     }

     const customer = await findOrCreateCustomer(parsed.data)

     // Store customer ID in cookie for booking flow
     cookies().set('customerId', customer.id, {
       httpOnly: true,
       secure: process.env.NODE_ENV === 'production',
       maxAge: 60 * 60 * 24 * 7, // 7 days
     })

     redirect(`/${parsed.data.locale}/booking`)
   }
   ```

1. Create app/api/availability/route.ts:
   ```typescript
   import { NextRequest, NextResponse } from 'next/server'
   import { prisma } from '@/lib/db/client'
   import { getAvailableSlots } from '@/lib/services/availability.service'

   export async function GET(request: NextRequest) {
     const { searchParams } = request.nextUrl
     const dateStr = searchParams.get('date') || new Date().toISOString().split('T')[0]
     const date = new Date(dateStr)

     // Get all active workers, resources, and existing bookings for the date
     const [workers, resources, bookings, service, schedules] = await Promise.all([
       prisma.worker.findMany({ where: { isActive: true } }),
       prisma.resource.findMany({ where: { isActive: true } }),
       prisma.booking.findMany({
         where: {
           startsAt: { gte: startOfDay(date), lt: endOfDay(date) },
           status: { in: ['CONFIRMED', 'PENDING'] },
         },
       }),
       prisma.service.findFirst({ where: { isActive: true } }),
       prisma.workerSchedule.findMany(),
     ])

     // Calculate available slots for each worker
     const slots = workers.flatMap(worker => {
       const workerSchedule = schedules.filter(s => s.workerId === worker.id)
       return getAvailableSlots(
         date,
         worker.id,
         resources,
         bookings.map(b => ({
           startTime: b.startsAt,
           endTime: b.endsAt,
           workerId: b.workerId,
           resourceId: b.resourceId,
         })),
         workerSchedule,
         service?.duration || 60
       ).map(slot => ({
         ...slot,
         workerId: worker.id,
         isAvailable: true,
       }))
     })

     return NextResponse.json({ workers, slots, date: dateStr })
   }
   ```

2. Create src/hooks/useAvailability.ts:

   ```typescript
   'use client'
   import useSWR from 'swr'

   const fetcher = (url: string) => fetch(url).then(r => r.json())

   export function useAvailability(date: string) {
     return useSWR(`/api/availability?date=${date}`, fetcher, {
       refreshInterval: 10000, // Poll every 10 seconds
       revalidateOnFocus: true,
     })
   }
   ```

3. Create app/actions/booking.ts:
   ```typescript
   'use server'

   import { redirect } from 'next/navigation'
   import { cookies } from 'next/headers'
   import { createBooking } from '@/lib/services/booking.service'
   import { prisma } from '@/lib/db/client'

   export async function createBookingAction(formData: FormData) {
     const customerId = cookies().get('customerId')?.value
     if (!customerId) {
       redirect('/register')
     }

     const workerId = formData.get('workerId') as string
     const time = formData.get('time') as string
     const date = formData.get('date') as string
     const resourceId = formData.get('resourceId') as string | null

     // Get default service
     const service = await prisma.service.findFirst({ where: { isActive: true } })
     if (!service) throw new Error('No service available')

     const startsAt = new Date(`${date}T${time}:00`)

     try {
       const booking = await createBooking({
         customerId,
         serviceId: service.id,
         workerId,
         resourceId: resourceId || undefined,
         startsAt,
       })

       redirect(`/booking/confirm?id=${booking.id}`)
     } catch (error) {
       return { error: (error as Error).message }
     }
   }
   ```
