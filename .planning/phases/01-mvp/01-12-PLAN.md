1. Create src/hooks/useCalendarPolling.ts:
   ```typescript
   'use client'

   import useSWR from 'swr'
   import { CalendarSlot, CalendarWorker, CalendarBooking } from '@/types/calendar'

   const fetcher = async (url: string) => {
     const res = await fetch(url)
     if (!res.ok) throw new Error('Failed to fetch calendar data')
     return res.json()
   }

   interface CalendarData {
     workers: CalendarWorker[]
     slots: CalendarSlot[]
     bookings?: CalendarBooking[] // Admin only
     date: string
   }

   interface UseCalendarOptions {
     date: string
     mode: 'user' | 'admin'
     pollingInterval?: number // milliseconds
   }

   export function useCalendarPolling({
     date,
     mode,
     pollingInterval = 10000, // Default 10 seconds
   }: UseCalendarOptions) {
     const endpoint = mode === 'user'
       ? `/api/availability?date=${date}`
       : `/api/admin/calendar?date=${date}`

     const { data, error, isLoading, mutate } = useSWR<CalendarData>(
       endpoint,
       fetcher,
       {
         refreshInterval: pollingInterval,
         revalidateOnFocus: true,
         revalidateOnReconnect: true,
         dedupingInterval: 2000,
         // Slower polling when tab not visible
         refreshWhenHidden: false,
         // Faster initial load
         revalidateOnMount: true,
       }
     )

     return {
       workers: data?.workers ?? [],
       slots: data?.slots ?? [],
       bookings: data?.bookings ?? [],
       date: data?.date ?? date,
       error,
       isLoading,
       refresh: mutate,
     }
   }

   // Optional: Adaptive polling based on user activity
   export function useAdaptivePolling(baseInterval: number = 10000) {
     const [interval, setInterval] = useState(baseInterval)

     useEffect(() => {
       let timeoutId: NodeJS.Timeout

       const handleActivity = () => {
         setInterval(baseInterval) // Fast polling on activity
         clearTimeout(timeoutId)
         timeoutId = setTimeout(() => {
           setInterval(baseInterval * 3) // Slow down after 30s inactivity
         }, 30000)
       }

       window.addEventListener('mousemove', handleActivity)
       window.addEventListener('keydown', handleActivity)

       return () => {
         window.removeEventListener('mousemove', handleActivity)
         window.removeEventListener('keydown', handleActivity)
         clearTimeout(timeoutId)
       }
     }, [baseInterval])

     return interval
   }
   ```

3. Create app/api/admin/calendar/route.ts for admin calendar data:
   ```typescript
   import { NextRequest, NextResponse } from 'next/server'
   import { prisma } from '@/lib/db/client'
   import { getAdminSession } from '@/lib/auth/admin'

   export async function GET(request: NextRequest) {
     const isAdmin = await getAdminSession()
     if (!isAdmin) {
       return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
     }

     const { searchParams } = request.nextUrl
     const dateStr = searchParams.get('date') || new Date().toISOString().split('T')[0]
     const date = new Date(dateStr)

     const startOfDay = new Date(date)
     startOfDay.setHours(0, 0, 0, 0)
     const endOfDay = new Date(date)
     endOfDay.setHours(23, 59, 59, 999)

     const [workers, bookings] = await Promise.all([
       prisma.worker.findMany({
         where: { isActive: true },
         orderBy: { name: 'asc' },
       }),
       prisma.booking.findMany({
         where: {
           startsAt: { gte: startOfDay, lt: endOfDay },
         },
         include: {
           customer: { select: { name: true, email: true, phone: true } },
           service: true,
           worker: true,
           resource: true,
         },
         orderBy: { startsAt: 'asc' },
       }),
     ])

     return NextResponse.json({
       workers,
       bookings: bookings.map(b => ({
         id: b.id,
         startsAt: b.startsAt,
         endsAt: b.endsAt,
         workerId: b.workerId,
         resourceId: b.resourceId,
         customerName: b.customer.name,
         customerEmail: b.customer.email,
         serviceName: b.service.name,
         status: b.status,
       })),
       date: dateStr,
     })
   }
   ```
