```typescript
/**
 * Generates CSV string from data array.
 * Adds UTF-8 BOM for Japanese Excel compatibility.
 * Escapes commas, quotes, and newlines in cell values.
 */
export function generateCSV(headers: string[], rows: string[][]): string {
  const BOM = '\uFEFF';  // UTF-8 BOM for Excel
  const escape = (val: string) => {
    if (val.includes(',') || val.includes('"') || val.includes('\n')) {
      return `"${val.replace(/"/g, '""')}"`;
    }
    return val;
  };
  const headerLine = headers.map(escape).join(',');
  const dataLines = rows.map(row => row.map(escape).join(','));
  return BOM + [headerLine, ...dataLines].join('\n');
}
```

   - Uses reporting service data
   - Headers: Period, Total Revenue, Bookings, New Customers, Existing Customers
   - Returns CSV string
